2.

分割字串，用來取得字串中想要的資訊。
fun main() {
    val word = "BlackiTeaime"
    val delim = "i"
 
    val list = word.split(delim)
 
    println(list)   
}

-> [Black,Tea,me]


3. 以下會用map來舉例在android中鍵值對的使用:

fun main() {
    
    val mana: MutableMap<String, Int> = mutableMapOf(Pair("Kyle", 100), Pair("Terry", 50), Pair("Lea", 5))
    
    for(key in mana.keys){  
        println("key = $key, value = ${mana.get(key)}")
    }

}

->
key = Kyle, value = 100
key = Terry, value = 50
key = Lea, value = 5
